*-- 전표번호 클릭과 동시에 전표조회화면 (FB03) 으로 이동하기

  SET PARAMETER ID 'GJR' FIELD gs_body-gjahr.   " Fiscal year 셋팅

  CALL TRANSACTION 'FB03' AND SKIP FIRST SCREEN.

*&---------------------------------------------------------------------*

*  <--  p2        text
*----------------------------------------------------------------------*
FORM get_main_data .

*DATA : BEGIN OF gs_body,
*         bukrs TYPE bkpf-bukrs,
*         gjahr TYPE bkpf-gjahr,
*         belnr TYPE bkpf-belnr,
*         blart TYPE bkpf-blart,
*         budat TYPE bkpf-budat,
*         bschl TYPE bsis-bschl,
*         mwskz TYPE bsis-mwskz,
*         werks TYPE bsis-werks,
*         name1 TYPE t001w-name1,   " for all entries 로 가지고 와야 함
*         gsber TYPE bsis-gsber,
*         gtext TYPE tgsbt-gtext,   " for all entries 로 가지고 와야 함
*         dmbtr TYPE bsik-dmbtr,
*         waers TYPE bkpf-waers,
*       END OF gs_body.
*DATA : gt_body LIKE TABLE OF gs_body.
*
**-- for NAME1
*DATA : gt_t001w TYPE TABLE OF t001w,
*       gs_t001w TYPE t001w.
*
**-- for GTEXT
*DATA : gt_tgsbt TYPE TABLE OF tgsbt,
*       gs_tgsbt TYPE tgsbt.

*-- backup table
  DATA : lt_body    LIKE gt_body WITH HEADER LINE.  " For backup (원본데이터 손실 막기 위해서)
  CLEAR : gs_body, gt_body.

  SELECT a~bukrs a~gjahr a~belnr a~blart a~budat b~bschl b~mwskz b~werks b~gsber b~dmbtr a~waers
    INTO CORRESPONDING FIELDS OF TABLE gt_body
    FROM bkpf AS a INNER JOIN bsis AS b
      ON a~bukrs EQ b~bukrs
     AND a~gjahr EQ b~gjahr
     AND a~belnr EQ b~belnr
    WHERE a~bukrs EQ p_bukrs
      AND a~gjahr EQ p_gjahr
      AND a~budat IN s_budat.

  IF gt_body IS INITIAL.
    MESSAGE s037 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

  CLEAR : lt_body, lt_body[].

*-- 0. gt_body 의 내용을 lt_body 로 복사
  MOVE gt_body TO lt_body[].

*-- 1. Sorting - Plant 기준 (werks)
  SORT lt_body[] BY werks ASCENDING.    " default => ascending

*-- 2. Delete duplicate values on the basis of lifnr
*-- COMPARING <field1> <field2> ...
*-- COMPARING ALL FIELDS => 전체 필드 기준
  DELETE ADJACENT DUPLICATES FROM lt_body COMPARING werks.

*-- 3. Get Plant Name (t001w-name1)
  CLEAR : gt_t001w.
  SELECT * INTO CORRESPONDING FIELDS OF TABLE gt_t001w
    FROM t001w
    FOR ALL ENTRIES IN lt_body[]
    WHERE werks EQ lt_body-werks.

*--------------------------------------------------------------------*

*-- 0. gt_body 의 내용을 lt_body 로 복사
  MOVE gt_body TO lt_body[].

*-- 1. Sorting  - Biz Area 기준 (gsber)
  SORT lt_body[] BY gsber ASCENDING.    " default => ascending

*-- 2. Delete duplicate values on the basis of hkont
*-- COMPARING <field1> <field2> ...
*-- COMPARING ALL FIELDS => 전체 필드 기준
  DELETE ADJACENT DUPLICATES FROM lt_body COMPARING gsber.

*-- 3. Get Biz Area Name (tgsbt-gtext)
  CLEAR : gt_tgsbt.
  SELECT * INTO CORRESPONDING FIELDS OF TABLE gt_tgsbt
    FROM tgsbt
    FOR ALL ENTRIES IN lt_body[]
    WHERE spras EQ sy-langu
    AND   gsber EQ lt_body-gsber.

ENDFORM.
*&---------------------------------------------------------------------*
*&      Form  MAKE_MAIN_BODY
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM make_main_body .

*-- backup table
  DATA : lv_tabix TYPE sy-tabix.

*-- Set Plant Name and Biz Area Name
  SORT gt_t001w BY werks ASCENDING.
  SORT gt_tgsbt BY gsber ASCENDING.
  LOOP AT gt_body INTO gs_body.

    lv_tabix = sy-tabix.

    CLEAR gs_t001w.
    READ TABLE gt_t001w INTO gs_t001w WITH KEY werks = gs_body-werks
                                    BINARY SEARCH.

    IF sy-subrc EQ 0.
      gs_body-name1 = gs_t001w-name1.
    ENDIF.

    CLEAR gs_tgsbt.
    READ TABLE gt_tgsbt INTO gs_tgsbt WITH KEY gsber = gs_body-gsber
                                      BINARY SEARCH.

    IF sy-subrc EQ 0.
      gs_body-gtext = gs_tgsbt-gtext.
    ENDIF.

    MODIFY gt_body FROM gs_body INDEX lv_tabix TRANSPORTING name1 gtext.

  ENDLOOP.

ENDFORM.

FORM data_save .

  DATA : lt_save   TYPE TABLE OF ztsmrt01_14 WITH HEADER LINE,
         lv_answer.

  CALL FUNCTION 'POPUP_TO_CONFIRM'
    EXPORTING
      titlebar              = '데이터 저장'
      text_question         = '이대로 저장하시겠습니까?'
      text_button_1         = 'Yes'(001)
      text_button_2         = 'No'(002)
      default_button        = '1'
      display_cancel_button = 'X'
    IMPORTING
      answer                = lv_answer.

  IF lv_answer NE '1'.
    EXIT.
  ENDIF.

**********************************************************************
* Save process
**********************************************************************
  CLEAR : lt_save, lt_save[].

  LOOP AT gt_body INTO gs_body.

    MOVE-CORRESPONDING gs_body TO lt_save.

*-- Set time stamp
    lt_save-erdat = sy-datum. " Current date
    lt_save-ernam = sy-uname. " Current user
    lt_save-erzet = sy-uzeit. " Current time

    APPEND lt_save.
    CLEAR lt_save.

  ENDLOOP.

*-- Save data
  MODIFY ztsmrt01_14 FROM TABLE lt_save[].
  COMMIT WORK.

*-- Show message
  MESSAGE s102.

ENDFORM.
