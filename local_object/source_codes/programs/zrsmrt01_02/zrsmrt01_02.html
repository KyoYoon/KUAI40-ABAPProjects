**********************************************************************
* INITIALIZATION : Set default value
**********************************************************************
INITIALIZATION.
*-- Set parameter value
  p_all = 'X'.


**********************************************************************
* AT SELECTION-SCREEN OUTPUT
**********************************************************************
AT SELECTION-SCREEN OUTPUT.
  LOOP AT SCREEN.

*-- 라디오 버튼 체크 (대상 결정)
    CASE 'X'.
      WHEN p_air.  " Show Airline ID & hide Connection ID
        IF screen-group1 EQ 'CAR'.  " MODIF ID => CAR (대문자로 표기)
          screen-active = 1.         " hide
        ENDIF.
        IF screen-group1 EQ 'CON'.  " MODIF ID => CON
          screen-active = 0.          " show
        ENDIF.
      WHEN p_con.  " Show Connection ID & hide Airline ID
        IF screen-group1 EQ 'CAR'.  " MODIF ID => CAR (대문자로 표기)
          screen-active = 0.         " show
        ENDIF.
        IF screen-group1 EQ 'CON'.  " MODIF ID => CON
          screen-active = 1.          " hide
        ENDIF.
      WHEN OTHERS.  " p_all : Display all
        IF screen-group1 EQ 'CAR'.  " MODIF ID => CAR (대문자로 표기)
          screen-active = 1.         " show
        ENDIF.
        IF screen-group1 EQ 'CON'.  " MODIF ID => CON
          screen-active = 1.          " show
        ENDIF.
    ENDCASE.

    MODIFY SCREEN.      " 바뀐 속성을 screen에 반영

  ENDLOOP.
