FORM get_base_data .

*-- Clear Internal Table With Header Line
  CLEAR : gt_body.

*-- Execute Open SQL
  SELECT a~carrid a~connid b~fldate a~airpfrom a~airpto b~planetype b~price
    INTO CORRESPONDING FIELDS OF TABLE gt_body
*-- inner join
    FROM spfli AS a INNER JOIN sflight AS b
    ON  a~carrid EQ b~carrid
    AND a~connid EQ b~connid
*-- End of join condition
    WHERE a~carrid EQ p_carrid
    AND   a~connid IN s_connid
    AND   b~fldate IN s_fldate.

ENDFORM.
