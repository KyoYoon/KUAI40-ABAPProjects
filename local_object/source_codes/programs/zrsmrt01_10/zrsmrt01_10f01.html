FORM get_base_data .
**********************************************************************
* Get airline info
**********************************************************************

  CLEAR: gt_scarr, gs_scarr.

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_scarr
    FROM scarr
    WHERE carrid IN s_carrid.


ENDFORM.
