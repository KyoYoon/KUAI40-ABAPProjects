INITIALIZATION.
*-- Set parameter value

  s_bookid-sign    = 'I'.      " 'I' : Include, 'E' : Exclude
  s_bookid-option  = 'BT'.     " Between
  s_bookid-low     = '200'.
  s_bookid-high    = '500'.

  APPEND s_bookid.     " header line 이 있는 internal table 이라서 왼쪽과 같이 작성 (Internal Table With Header Line)

*-- Set parameter value
  p_aa = 'X'.
  p_dl = 'X'.

  p_air = 'X'.


**********************************************************************
* AT SELECTION-SCREEN OUTPUT
**********************************************************************
AT SELECTION-SCREEN OUTPUT.
  LOOP AT SCREEN.

*-- 라디오 버튼 체크 (대상 결정)
    CASE 'X'.
      WHEN p_air.  "
        IF screen-group1 EQ 'CAR'.  " MODIF ID => CAR (대문자로 표기)
          screen-input = 0.         " read-only mode
        ENDIF.
        IF screen-group1 EQ 'CLA'.  " MODIF ID => CLA
          screen-input = 1.          " edit mode
        ENDIF.
        IF screen-group1 EQ 'BOO'.  " MODIF ID => BOO
          screen-input = 1.          " edit mode
        ENDIF.
      WHEN p_cla.  "
        IF screen-group1 EQ 'CAR'.  " MODIF ID => CAR (대문자로 표기)
          screen-input = 1.         " edit mode
        ENDIF.
        IF screen-group1 EQ 'CLA'.  " MODIF ID => CLA
          screen-input = 0.          " read-only mode
        ENDIF.
        IF screen-group1 EQ 'BOO'.  " MODIF ID => BOO
          screen-input = 1.          " edit mode
        ENDIF.
      WHEN p_boo.  "
        IF screen-group1 EQ 'CAR'.  " MODIF ID => CAR (대문자로 표기)
          screen-input = 1.         " edit mode
        ENDIF.
        IF screen-group1 EQ 'CLA'.  " MODIF ID => CLA
          screen-input = 1.          " edit mode
        ENDIF.
        IF screen-group1 EQ 'BOO'.  " MODIF ID => BOO
          screen-input = 0.          " read-only mode
        ENDIF.
      WHEN OTHERS.  " p_all : Display all
        IF screen-group1 EQ 'CAR'.  " MODIF ID => CAR (대문자로 표기)
          screen-input = 1.         " edit mode
        ENDIF.
        IF screen-group1 EQ 'CLA'.  " MODIF ID => CLA
          screen-input = 1.          " edit mode
        ENDIF.
        IF screen-group1 EQ 'BOO'.  " MODIF ID => BOO
          screen-input = 1.          " edit mode
        ENDIF.
    ENDCASE.

    MODIFY SCREEN.      " 바뀐 속성을 screen에 반영

  ENDLOOP.
